    let money = 1000,
        round = 0,
        choice = "tai";

    const BET = 100;

    const rotations = {
        1: "rotateX(0deg) rotateY(0deg)",
        2: "rotateX(-90deg)",
        3: "rotateY(90deg)",
        4: "rotateY(-90deg)",
        5: "rotateX(90deg)",
        6: "rotateY(180deg)"
    };

    ["d1", "d2", "d3"].forEach(id => {
        document.getElementById(id).innerHTML = `
            <div class="face front">1</div>
            <div class="face back">6</div>

            <div class="face left">4</div>

    function choose(c) {
        choice = c;
        taiBox.classList.remove("active");
        xiuBox.classList.remove("active");
        document.getElementById(c + "Box").classList.add("active");
    }

    function rollOne(id, val) {
        const el = document.getElementById(id);
        el.style.transform = `rotateX(${Math.random() * 720}deg) rotateY(${Math.random() * 720}deg)`;
        setTimeout(() => el.style.transform = rotations[val], 600);
    }

    function play() {
        if (money <= 0) return;

        round++;
        document.getElementById("round").innerText = round;

        let d1 = r(), d2 = r(), d3 = r();
        let total = d1 + d2 + d3;

        rollOne("d1", d1);
        rollOne("d2", d2);
        rollOne("d3", d3);

        let winRate = round <= 3 ? 0.8 : round <= 6 ? 0.5 : 0.1;
        let win = Math.random() < winRate &&
            ((choice === "tai" && total >= 11) || (choice === "xiu" && total <= 10));

        money += win ? BET : -BET;
        document.getElementById("money").innerText = money;

        if (money <= 0) {
            setTimeout(() => {
                document.getElementById("overlay").classList.add("active");
            }, 1200);
        }
    }

    function r() {
        return Math.floor(Math.random() * 6) + 1;
    }
